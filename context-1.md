const http = require('http');

function makeRequest(method, path, data = null, token = null) {
  return new Promise((resolve, reject) => {
    const postData = data ? JSON.stringify(data) : null;
    const options = {
      hostname: 'localhost',
      port: 3002,
      path: path,
      method: method,
      headers: {
        'Content-Type': 'application/json',
      }
    };
    if (token) {
      options.headers['Authorization'] = 'Bearer ' + token;
    }
    if (postData) {
      options.headers['Content-Length'] = Buffer.byteLength(postData);
    }
    const req = http.request(options, (res) => {
      let body = '';
      res.on('data', (chunk) => body += chunk);
      res.on('end', () => {
        try {
          resolve(JSON.parse(body));
        } catch (e) {
          resolve(body);
        }
      });
    });
    req.on('error', reject);
    if (postData) req.write(postData);
    req.end();
  });
}

## 项目概览

### 技术栈
- **前端**: React 18 + Vite + React Router + Zustand + Axios
- **后端**: Express.js + SQLite (better-sqlite3)
- **认证**: JWT + bcryptjs
- **文件上传**: Multer

### 已实现功能

**用户系统**
- 注册/登录页面
- 个人中心（5个Tab页）

**首页模块**
- 瀑布流展示衣物
- 分类筛选（上衣、裤子、裙子、外套、鞋子、配饰、其他）
- 尺码筛选
- 品牌搜索
- 关键词搜索
- 心愿单匹配推荐提示

**发布模块**
- 多图上传（最多9张）
- 品牌、分类、尺码、成色选择
- 详细描述
- 期望交换的衣物类型

**详情页**
- 多图轮播展示
- 衣物详细信息
- 卖家信息
- "我想换"发起交换申请弹窗

**交换流程**
- 发起交换申请（选择自己的衣物交换）
- 收到的申请列表（通过/拒绝）
- 发出的申请列表
- 交换记录
- 双方确认完成
- 互相评价（五星评分 + 文字评价）

**心愿单功能**
- 添加心愿（分类、品牌、尺码、关键词）
- 首页推荐匹配的新衣物

**管理员后台**
- 内容审核（通过/驳回）
- 数据统计（用户数、衣物数、交换数）
- 热门品类排行
- 热门品牌排行
- 近30天交换量图表
